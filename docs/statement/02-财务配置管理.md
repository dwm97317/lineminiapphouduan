# 财务配置管理

**负责人**：后端开发A + 前端开发A  
**预计工期**：2天  
**依赖**：数据库设计完成

## 一、功能概述

财务配置管理提供独立的配置页面，用于管理：
1. 账单模板配置（LOGO、标题、收款二维码、提示文字）
2. 全局默认单价配置（固定单价）
3. 客户单价配置（5种计价方式）

**页面位置**：系统设置 → 财务配置

## 二、后端开发

### 2.1 模型层（FinanceConfig.php）

**文件路径**：`source/application/common/model/FinanceConfig.php`

```php
<?php
namespace app\common\model;

use think\Model;

class FinanceConfig extends Model
{
    protected $name = 'finance_config';
    protected $pk = 'id';
    
    // 配置类型常量
    const TYPE_TEMPLATE = 1;  // 账单模板
    const TYPE_CUSTOMER_PRICE = 2;  // 客户单价
    const TYPE_GLOBAL_PRICE = 3;  // 全局默认单价
    const TYPE_HISTORY_PRICE = 4;  // 历史单价表
    
    // 计价方式常量
    const PRICE_TYPE_FIXED = 1;  // 固定单价
    const PRICE_TYPE_TIER = 2;  // 阶梯价格
    const PRICE_TYPE_LINE = 3;  // 按线路
    const PRICE_TYPE_RANGE = 4;  // 按重量区间
    const PRICE_TYPE_FORMULA = 5;  // 自定义公式
    
    /**
     * 获取默认账单模板
     */
    public static function getDefaultTemplate()
    {
        return self::where('config_type', self::TYPE_TEMPLATE)
            ->where('is_default', 1)
            ->where('status', 1)
            ->find();
    }
    
    /**
     * 获取客户单价配置
     */
    public static function getPriceByMember($memberId)
    {
        return self::where('config_type', self::TYPE_CUSTOMER_PRICE)
            ->where('member_id', $memberId)
            ->where('status', 1)
            ->find();
    }
    
    /**
     * 获取全局默认单价配置
     */
    public static function getGlobalDefaultPrice()
    {
        return self::where('config_type', self::TYPE_GLOBAL_PRICE)
            ->where('member_id', null)
            ->where('status', 1)
            ->find();
    }
    
    /**
     * 获取历史单价表配置
     */
    public static function getHistoryPriceTable()
    {
        return self::where('config_type', self::TYPE_HISTORY_PRICE)
            ->where('status', 1)
            ->find();
    }
    
    /**
     * 从历史单价表获取客户单价
     */
    public static function getPriceFromHistory($memberId)
    {
        $historyTable = self::getHistoryPriceTable();
        
        if (!$historyTable || empty($historyTable['price_data_json'])) {
            return null;
        }
        
        $priceData = json_decode($historyTable['price_data_json'], true);
        
        if (isset($priceData[$memberId])) {
            return [
                'price_type' => self::PRICE_TYPE_FIXED,
                'unit_price' => $priceData[$memberId]
            ];
        }
        
        return null;
    }
    
    /**
     * 获取有效单价配置（带回退逻辑）
     */
    public static function getEffectivePrice($memberId)
    {
        // 1. 优先客户专属配置
        $config = self::getPriceByMember($memberId);
        
        if ($config) {
            return $config;
        }
        
        // 2. 回退到历史单价表
        $historyPrice = self::getPriceFromHistory($memberId);
        
        if ($historyPrice) {
            return $historyPrice;
        }
        
        // 3. 回退到全局默认
        $defaultConfig = self::getGlobalDefaultPrice();
        
        if ($defaultConfig) {
            return $defaultConfig;
        }
        
        // 4. 系统兜底：返回固定单价46元/KG
        return [
            'price_type' => self::PRICE_TYPE_FIXED,
            'unit_price' => 46.00
        ];
    }
}
```

### 2.2 服务层（FinanceConfigService.php）

**文件路径**：`source/application/common/service/FinanceConfigService.php`

```php
<?php
namespace app\common\service;

use app\common\model\FinanceConfig;
use app\common\library\helper;

class FinanceConfigService
{
    private $wxapp_id;
    
    public function __construct($wxapp_id = null)
    {
        $this->wxapp_id = $wxapp_id ?: helper::getWxappId();
    }
    
    /**
     * 获取账单模板列表
     */
    public function getTemplateList($params = [])
    {
        $model = FinanceConfig::where('config_type', FinanceConfig::TYPE_TEMPLATE)
            ->where('wxapp_id', $this->wxapp_id);
        
        return $model->order('is_default DESC, sort ASC, id DESC')
            ->paginate(15);
    }
    
    /**
     * 保存账单模板
     */
    public function saveTemplate($data)
    {
        // 如果设为默认，取消其他默认模板
        if ($data['is_default'] == 1) {
            FinanceConfig::where('config_type', FinanceConfig::TYPE_TEMPLATE)
                ->where('wxapp_id', $this->wxapp_id)
                ->update(['is_default' => 0]);
        }
        
        $data['config_type'] = FinanceConfig::TYPE_TEMPLATE;
        $data['wxapp_id'] = $this->wxapp_id;
        
        if (isset($data['id']) && $data['id'] > 0) {
            return FinanceConfig::update($data);
        } else {
            return FinanceConfig::create($data);
        }
    }
    
    /**
     * 获取全局默认单价
     */
    public function getGlobalPrice()
    {
        return FinanceConfig::where('config_type', FinanceConfig::TYPE_GLOBAL_PRICE)
            ->where('wxapp_id', $this->wxapp_id)
            ->find();
    }
    
    /**
     * 保存全局默认单价
     */
    public function saveGlobalPrice($unitPrice)
    {
        $config = $this->getGlobalPrice();
        
        $data = [
            'config_type' => FinanceConfig::TYPE_GLOBAL_PRICE,
            'config_name' => '全局默认单价',
            'member_id' => null,
            'price_type' => FinanceConfig::PRICE_TYPE_FIXED,
            'unit_price' => $unitPrice,
            'is_default' => 1,
            'status' => 1,
            'wxapp_id' => $this->wxapp_id
        ];
        
        if ($config) {
            $data['id'] = $config['id'];
            return FinanceConfig::update($data);
        } else {
            return FinanceConfig::create($data);
        }
    }
    
    /**
     * 获取客户单价列表
     */
    public function getCustomerPriceList($params = [])
    {
        $model = FinanceConfig::where('config_type', FinanceConfig::TYPE_CUSTOMER_PRICE)
            ->where('wxapp_id', $this->wxapp_id);
        
        // 搜索客户
        if (!empty($params['member_id'])) {
            $model->where('member_id', $params['member_id']);
        }
        
        return $model->order('id DESC')->paginate(15);
    }
    
    /**
     * 保存客户单价配置
     */
    public function saveCustomerPrice($data)
    {
        // 验证客户是否已存在配置
        if (!isset($data['id'])) {
            $exists = FinanceConfig::where('config_type', FinanceConfig::TYPE_CUSTOMER_PRICE)
                ->where('member_id', $data['member_id'])
                ->where('wxapp_id', $this->wxapp_id)
                ->find();
            
            if ($exists) {
                throw new \Exception('该客户已存在单价配置');
            }
        }
        
        $data['config_type'] = FinanceConfig::TYPE_CUSTOMER_PRICE;
        $data['wxapp_id'] = $this->wxapp_id;
        
        // 根据计价方式处理JSON字段
        switch ($data['price_type']) {
            case FinanceConfig::PRICE_TYPE_TIER:
                $data['price_tier_json'] = json_encode($data['tiers']);
                break;
            case FinanceConfig::PRICE_TYPE_LINE:
                $data['price_line_json'] = json_encode($data['lines']);
                break;
            case FinanceConfig::PRICE_TYPE_RANGE:
                $data['price_range_json'] = json_encode($data['ranges']);
                break;
            case FinanceConfig::PRICE_TYPE_FORMULA:
                $data['price_formula'] = json_encode(['formula' => $data['formula']]);
                break;
        }
        
        if (isset($data['id']) && $data['id'] > 0) {
            return FinanceConfig::update($data);
        } else {
            return FinanceConfig::create($data);
        }
    }
    
    /**
     * 删除配置
     */
    public function deleteConfig($id)
    {
        $config = FinanceConfig::find($id);
        
        if (!$config || $config['wxapp_id'] != $this->wxapp_id) {
            throw new \Exception('配置不存在');
        }
        
        // 不允许删除默认模板
        if ($config['config_type'] == FinanceConfig::TYPE_TEMPLATE && $config['is_default'] == 1) {
            throw new \Exception('不能删除默认模板');
        }
        
        // 不允许删除全局默认单价
        if ($config['config_type'] == FinanceConfig::TYPE_GLOBAL_PRICE) {
            throw new \Exception('不能删除全局默认单价');
        }
        
        return $config->delete();
    }
}
```

### 2.3 控制器层（FinanceConfigController.php）

**文件路径**：`source/application/store/controller/finance/Config.php`

```php
<?php
namespace app\store\controller\finance;

use app\store\controller\Controller;
use app\common\service\FinanceConfigService;

class Config extends Controller
{
    /**
     * 财务配置主页
     */
    public function index()
    {
        return $this->fetch();
    }
    
    /**
     * 获取账单模板列表
     */
    public function templateList()
    {
        $service = new FinanceConfigService();
        $list = $service->getTemplateList($this->request->param());
        
        return $this->renderSuccess('', compact('list'));
    }
    
    /**
     * 保存账单模板
     */
    public function saveTemplate()
    {
        $data = $this->request->post();
        $service = new FinanceConfigService();
        
        if ($service->saveTemplate($data)) {
            return $this->renderSuccess('保存成功');
        }
        
        return $this->renderError('保存失败');
    }
    
    /**
     * 获取全局默认单价
     */
    public function getGlobalPrice()
    {
        $service = new FinanceConfigService();
        $config = $service->getGlobalPrice();
        
        return $this->renderSuccess('', [
            'unit_price' => $config ? $config['unit_price'] : 46.00
        ]);
    }
    
    /**
     * 保存全局默认单价
     */
    public function saveGlobalPrice()
    {
        $unitPrice = $this->request->post('unit_price');
        
        if (empty($unitPrice) || $unitPrice <= 0) {
            return $this->renderError('请输入有效的单价');
        }
        
        $service = new FinanceConfigService();
        
        if ($service->saveGlobalPrice($unitPrice)) {
            return $this->renderSuccess('保存成功');
        }
        
        return $this->renderError('保存失败');
    }
    
    /**
     * 获取客户单价列表
     */
    public function customerPriceList()
    {
        $service = new FinanceConfigService();
        $list = $service->getCustomerPriceList($this->request->param());
        
        return $this->renderSuccess('', compact('list'));
    }
    
    /**
     * 保存客户单价配置
     */
    public function saveCustomerPrice()
    {
        $data = $this->request->post();
        $service = new FinanceConfigService();
        
        try {
            if ($service->saveCustomerPrice($data)) {
                return $this->renderSuccess('保存成功');
            }
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
        
        return $this->renderError('保存失败');
    }
    
    /**
     * 删除配置
     */
    public function delete()
    {
        $id = $this->request->post('id');
        $service = new FinanceConfigService();
        
        try {
            if ($service->deleteConfig($id)) {
                return $this->renderSuccess('删除成功');
            }
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
        
        return $this->renderError('删除失败');
    }
    
    /**
     * 导入历史单价表
     */
    public function importHistoryPrice()
    {
        $file = $this->request->file('file');
        
        if (!$file) {
            return $this->renderError('请选择文件');
        }
        
        $service = new FinanceConfigService();
        
        try {
            $result = $service->importHistoryPriceTable($file);
            
            return $this->renderSuccess('导入成功', [
                'total_count' => count(json_decode($result['price_data_json'], true))
            ]);
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
    }
    
    /**
     * 获取历史单价表信息
     */
    public function getHistoryPriceTable()
    {
        $service = new FinanceConfigService();
        $config = $service->getHistoryPriceTable();
        
        return $this->renderSuccess('', compact('config'));
    }
    
    /**
     * 上传图片
     */
    public function uploadImage()
    {
        // 使用系统已有的图片上传功能
        $UploadService = new \app\common\service\UploadService();
        
        if (!$UploadService->upload()) {
            return $this->renderError($UploadService->getError());
        }
        
        return $this->renderSuccess('上传成功', [
            'file_id' => $UploadService->getFileId(),
            'file_path' => $UploadService->getFilePath()
        ]);
    }
}
```

## 四、API接口文档

### 4.1 账单模板接口

**获取模板列表**
- URL: `/store/finance.config/templateList`
- Method: GET
- Response: `{code: 1, data: {list: [...]}}`

**保存模板**
- URL: `/store/finance.config/saveTemplate`
- Method: POST
- Params: `{config_name, logo_path, title, alipay_qr_path, wechat_qr_path, ...}`

### 4.2 全局单价接口

**获取全局单价**
- URL: `/store/finance.config/getGlobalPrice`
- Method: GET
- Response: `{code: 1, data: {unit_price: 46.00}}`

**保存全局单价**
- URL: `/store/finance.config/saveGlobalPrice`
- Method: POST
- Params: `{unit_price: 46.00}`

### 4.3 客户单价接口

**获取客户单价列表**
- URL: `/store/finance.config/customerPriceList`
- Method: GET
- Response: `{code: 1, data: {list: [...]}}`

**保存客户单价**
- URL: `/store/finance.config/saveCustomerPrice`
- Method: POST
- Params: `{member_id, price_type, unit_price, tiers, ...}`

### 4.4 通用接口

**删除配置**
- URL: `/store/finance.config/delete`
- Method: POST
- Params: `{id}`

**上传图片**
- URL: `/store/finance.config/uploadImage`
- Method: POST
- Params: `FormData(file)`
