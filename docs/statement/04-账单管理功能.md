# 账单管理功能

**负责人**：后端开发C + 前端开发B  
**预计工期**：1-2天  
**依赖**：数据库设计、账单生成功能

## 一、功能概述

账单管理功能提供对已生成账单的查看、支付标记、作废等操作，主要包括：
1. 集运订单列表页扩展（显示账单信息）
2. 账单详情查看
3. 账单支付标记
4. 账单作废
5. 账单Excel下载

**页面位置**：集运订单管理 → 订单列表

## 二、业务流程

### 2.1 账单生命周期

```
[草稿状态] → [已确认] → [已发送] → [已支付]
   ↓            ↓          ↓          ↓
 可编辑       可作废     可作废     不可作废
 可删除       可支付     可支付     已完成
```

### 2.2 支付标记流程

```
1. 财务发送账单给客户
   ↓
2. 客户付款
   ↓
3. 财务在系统中找到对应账单
   ↓
4. 点击"标记为已支付"
   ↓
5. 系统更新：
   - 账单pay_status = 2
   - 账单pay_time = 当前时间
   - 所有关联订单is_pay = 1
   ↓
6. 完成
```

### 2.3 账单作废流程

```
1. 财务发现账单有误
   ↓
2. 点击"作废账单"
   ↓
3. 填写作废原因
   ↓
4. 系统更新：
   - 账单status = 已作废
   - 解除订单绑定（statement_id = NULL）
   - 订单恢复到待出账状态
   ↓
5. 可重新生成账单
```

### 3.1 模型层扩展（Statement.php）

**文件路径**：`source/application/common/model/Statement.php`

```php
<?php
namespace app\common\model;

use think\Model;

class Statement extends Model
{
    // ... 已有代码 ...
    
    // 账单状态常量
    const STATUS_DRAFT = 1;  // 草稿
    const STATUS_CONFIRMED = 2;  // 已确认
    const STATUS_SENT = 3;  // 已发送
    const STATUS_VOIDED = 9;  // 已作废
    
    /**
     * 标记为已支付
     */
    public function markAsPaid($remark = '')
    {
        if ($this['pay_status'] == self::PAY_STATUS_PAID) {
            throw new \Exception('账单已支付，无需重复操作');
        }
        
        \think\Db::startTrans();
        try {
            // 1. 更新账单状态
            $this->save([
                'pay_status' => self::PAY_STATUS_PAID,
                'pay_time' => date('Y-m-d H:i:s'),
                'pay_remark' => $remark
            ]);
            
            // 2. 更新所有关联订单的支付状态
            Package::where('statement_id', $this['id'])
                ->update([
                    'is_pay' => 1,
                    'pay_time' => date('Y-m-d H:i:s')
                ]);
            
            \think\Db::commit();
            return true;
            
        } catch (\Exception $e) {
            \think\Db::rollback();
            throw $e;
        }
    }
    
    /**
     * 作废账单
     */
    public function voidStatement($reason = '')
    {
        if ($this['pay_status'] == self::PAY_STATUS_PAID) {
            throw new \Exception('已支付的账单不能作废');
        }
        
        \think\Db::startTrans();
        try {
            // 1. 更新账单状态
            $this->save([
                'status' => self::STATUS_VOIDED,
                'void_reason' => $reason,
                'void_time' => date('Y-m-d H:i:s')
            ]);
            
            // 2. 解除订单绑定
            Package::where('statement_id', $this['id'])
                ->update(['statement_id' => null]);
            
            \think\Db::commit();
            return true;
            
        } catch (\Exception $e) {
            \think\Db::rollback();
            throw $e;
        }
    }
    
    /**
     * 获取账单统计信息
     */
    public function getStatistics()
    {
        return [
            'total_packages' => $this['total_packages'],
            'total_weight' => $this['total_weight'],
            'total_amount' => $this['total_amount'],
            'paid_amount' => $this['pay_status'] == self::PAY_STATUS_PAID ? $this['total_amount'] : 0,
            'unpaid_amount' => $this['pay_status'] == self::PAY_STATUS_UNPAID ? $this['total_amount'] : 0
        ];
    }
}
```

### 3.2 服务层（StatementManageService.php）

**文件路径**：`source/application/common/service/StatementManageService.php`

```php
<?php
namespace app\common\service;

use app\common\model\Statement;
use app\common\model\Package;

class StatementManageService
{
    private $wxapp_id;
    
    public function __construct($wxapp_id = null)
    {
        $this->wxapp_id = $wxapp_id ?: helper::getWxappId();
    }
    
    /**
     * 获取账单列表
     */
    public function getStatementList($params = [])
    {
        $model = Statement::where('wxapp_id', $this->wxapp_id);
        
        // 客户筛选
        if (!empty($params['member_id'])) {
            $model->where('member_id', $params['member_id']);
        }
        
        // 支付状态筛选
        if (isset($params['pay_status']) && $params['pay_status'] !== '') {
            $model->where('pay_status', $params['pay_status']);
        }
        
        // 账单状态筛选
        if (isset($params['status']) && $params['status'] !== '') {
            $model->where('status', $params['status']);
        }
        
        // 日期范围
        if (!empty($params['start_date'])) {
            $model->where('start_date', '>=', $params['start_date']);
        }
        if (!empty($params['end_date'])) {
            $model->where('end_date', '<=', $params['end_date']);
        }
        
        // 账单编号搜索
        if (!empty($params['statement_no'])) {
            $model->where('statement_no', 'like', '%' . $params['statement_no'] . '%');
        }
        
        return $model->order('id', 'desc')->paginate(15);
    }
    
    /**
     * 获取账单详情
     */
    public function getStatementDetail($statementId)
    {
        $statement = Statement::where('id', $statementId)
            ->where('wxapp_id', $this->wxapp_id)
            ->find();
        
        if (!$statement) {
            throw new \Exception('账单不存在');
        }
        
        // 获取关联订单
        $packages = Package::where('statement_id', $statementId)
            ->where('is_delete', 0)
            ->select();
        
        // 获取统计信息
        $statistics = $statement->getStatistics();
        
        return [
            'statement' => $statement,
            'packages' => $packages,
            'statistics' => $statistics
        ];
    }
    
    /**
     * 标记账单为已支付
     */
    public function markStatementAsPaid($statementId, $remark = '')
    {
        $statement = Statement::where('id', $statementId)
            ->where('wxapp_id', $this->wxapp_id)
            ->find();
        
        if (!$statement) {
            throw new \Exception('账单不存在');
        }
        
        return $statement->markAsPaid($remark);
    }
    
    /**
     * 作废账单
     */
    public function voidStatement($statementId, $reason = '')
    {
        $statement = Statement::where('id', $statementId)
            ->where('wxapp_id', $this->wxapp_id)
            ->find();
        
        if (!$statement) {
            throw new \Exception('账单不存在');
        }
        
        if (empty($reason)) {
            throw new \Exception('请填写作废原因');
        }
        
        return $statement->voidStatement($reason);
    }
    
    /**
     * 下载账单Excel
     */
    public function downloadStatementExcel($statementId)
    {
        $statement = Statement::where('id', $statementId)
            ->where('wxapp_id', $this->wxapp_id)
            ->find();
        
        if (!$statement) {
            throw new \Exception('账单不存在');
        }
        
        if (empty($statement['excel_path']) || !file_exists($statement['excel_path'])) {
            throw new \Exception('Excel文件不存在');
        }
        
        return $statement['excel_path'];
    }
    
    /**
     * 获取客户的账单历史
     */
    public function getMemberStatementHistory($memberId, $limit = 10)
    {
        return Statement::where('member_id', $memberId)
            ->where('wxapp_id', $this->wxapp_id)
            ->where('status', '<>', Statement::STATUS_VOIDED)
            ->order('id', 'desc')
            ->limit($limit)
            ->select();
    }
}
```

### 3.3 控制器层（StatementController.php）

**文件路径**：`source/application/store/controller/package/Statement.php`

```php
<?php
namespace app\store\controller\package;

use app\store\controller\Controller;
use app\common\service\StatementManageService;

class Statement extends Controller
{
    /**
     * 账单列表
     */
    public function index()
    {
        $service = new StatementManageService();
        $list = $service->getStatementList($this->request->param());
        
        return $this->renderSuccess('', compact('list'));
    }
    
    /**
     * 账单详情
     */
    public function detail()
    {
        $statementId = $this->request->param('statement_id');
        
        if (empty($statementId)) {
            return $this->renderError('缺少参数');
        }
        
        try {
            $service = new StatementManageService();
            $data = $service->getStatementDetail($statementId);
            
            return $this->renderSuccess('', $data);
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
    }
    
    /**
     * 标记为已支付
     */
    public function markAsPaid()
    {
        $statementId = $this->request->post('statement_id');
        $remark = $this->request->post('remark', '');
        
        if (empty($statementId)) {
            return $this->renderError('缺少参数');
        }
        
        try {
            $service = new StatementManageService();
            
            if ($service->markStatementAsPaid($statementId, $remark)) {
                return $this->renderSuccess('标记成功');
            }
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
        
        return $this->renderError('标记失败');
    }
    
    /**
     * 作废账单
     */
    public function voidStatement()
    {
        $statementId = $this->request->post('statement_id');
        $reason = $this->request->post('reason', '');
        
        if (empty($statementId)) {
            return $this->renderError('缺少参数');
        }
        
        try {
            $service = new StatementManageService();
            
            if ($service->voidStatement($statementId, $reason)) {
                return $this->renderSuccess('作废成功');
            }
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
        
        return $this->renderError('作废失败');
    }
    
    /**
     * 下载Excel
     */
    public function downloadExcel()
    {
        $statementId = $this->request->param('statement_id');
        
        if (empty($statementId)) {
            return $this->renderError('缺少参数');
        }
        
        try {
            $service = new StatementManageService();
            $filePath = $service->downloadStatementExcel($statementId);
            
            // 下载文件
            $fileName = basename($filePath);
            header('Content-Type: application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
            header('Content-Disposition: attachment;filename="' . $fileName . '"');
            header('Cache-Control: max-age=0');
            
            readfile($filePath);
            exit;
            
        } catch (\Exception $e) {
            return $this->renderError($e->getMessage());
        }
    }
    
    /**
     * 获取客户账单历史
     */
    public function memberHistory()
    {
        $memberId = $this->request->param('member_id');
        
        if (empty($memberId)) {
            return $this->renderError('缺少参数');
        }
        
        $service = new StatementManageService();
        $history = $service->getMemberStatementHistory($memberId);
        
        return $this->renderSuccess('', compact('history'));
    }
}
```

### 4.3 JavaScript交互

```javascript
<script>
// 查看账单详情
$(document).on('click', '.view-statement', function() {
    var statementId = $(this).data('id');
    
    $.ajax({
        url: '/store/package.statement/detail',
        data: { statement_id: statementId },
        success: function(res) {
            if (res.code == 1) {
                showStatementDetail(res.data);
            } else {
                layer.msg(res.msg);
            }
        }
    });
});

// 显示账单详情
function showStatementDetail(data) {
    var statement = data.statement;
    var packages = data.packages;
    
    // 填充基本信息
    $('#statement_no').text(statement.statement_no);
    $('#member_name').text(statement.member_name);
    $('#date_range').text(statement.start_date + ' 至 ' + statement.end_date);
    $('#total_packages').text(statement.total_packages);
    $('#total_weight').text(statement.total_weight);
    $('#total_amount').text('¥' + statement.total_amount);
    
    // 支付状态
    if (statement.pay_status == 2) {
        $('#pay_status_badge').html('<span class="label label-success">已支付</span>');
        $('#pay_time_row').show();
        $('#pay_time').text(statement.pay_time);
        $('#btn-mark-paid').hide();
        $('#btn-void').hide();
    } else {
        $('#pay_status_badge').html('<span class="label label-warning">未支付</span>');
        $('#pay_time_row').hide();
        $('#btn-mark-paid').show();
        $('#btn-void').show();
    }
    
    // 填充订单列表
    var html = '';
    $.each(packages, function(i, pkg) {
        html += '<tr>';
        html += '<td>' + pkg.express_num + '</td>';
        html += '<td>' + pkg.weight + '</td>';
        html += '<td>' + (pkg.unit_price || '-') + '</td>';
        html += '<td>' + (pkg.amount || pkg.real_payment) + '</td>';
        html += '<td>' + pkg.entering_warehouse_time + '</td>';
        html += '</tr>';
    });
    $('#package_list').html(html);
    
    // 保存账单ID
    $('#statementDetailModal').data('statement-id', statement.id);
    
    // 显示弹窗
    $('#statementDetailModal').modal('show');
}

// 标记为已支付
```
