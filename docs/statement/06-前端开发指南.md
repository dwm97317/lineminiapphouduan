## 一、技术栈

### 1.1 核心技术
- **jQuery**：JavaScript库，简化DOM操作和Ajax请求
- **Amaze UI**：国产前端框架，提供响应式布局和UI组件
- **ECharts**：数据可视化图表库
- **Datetimepicker**：日期时间选择插件
- **IconFont**：阿里巴巴矢量图标库

### 1.2 架构说明
- **后端**：PHP (ThinkPHP 5.0框架)
- **前端**：HTML + CSS + JavaScript
- **模板引擎**：PHP原生模板（.php文件）
- **样式**：Amaze UI + 自定义CSS

### 1.3 开发工具
- Chrome DevTools
- VS Code
- Git

// 加载模板列表
function loadTemplateList() {
    $.ajax({
        url: '/store/finance.config/templateList',
        success: function(res) {
            if (res.code == 1) {
                renderTemplateList(res.data.list);
            }
        }
    });

// 渲染模板列表
function renderTemplateList(list) {
    var html = '';
    $.each(list, function(i, item) {
        html += '<tr>';
        html += '<td>' + item.id + '</td>';
        html += '<td>' + item.config_name + '</td>';
        html += '<td>' + item.title + '</td>';
        html += '<td>';
        if (item.logo_path) {
            html += '<img src="' + item.logo_path + '" height="30">';
        }
        html += '</td>';
        html += '<td>';
        if (item.is_default == 1) {
            html += '<span class="label label-success">默认</span>';
        }
        html += '</td>';
        html += '<td>';
        html += '<span class="label label-' + (item.status == 1 ? 'success' : 'default') + '">';
        html += item.status == 1 ? '启用' : '禁用';
        html += '</span>';
        html += '</td>';
        html += '<td>';
        html += '<button class="btn btn-xs btn-primary" onclick="editTemplate(' + item.id + ')">编辑</button> ';
        if (item.is_default != 1) {
            html += '<button class="btn btn-xs btn-danger" onclick="deleteTemplate(' + item.id + ')">删除</button>';
        }
        html += '</td>';
        html += '</tr>';
    });
    $('#template-list').html(html);
}

function editTemplate(id) {
    $.ajax({
        url: '/store/finance.config/getTemplate',
        data: { id: id },
        success: function(res) {
            if (res.code == 1) {
                var data = res.data;
                $('#template_id').val(data.id);
                $('[name="config_name"]').val(data.config_name);
                $('[name="title"]').val(data.title);
                $('[name="logo_path"]').val(data.logo_path);
                $('[name="alipay_qr_path"]').val(data.alipay_qr_path);
                $('[name="wechat_qr_path"]').val(data.wechat_qr_path);
                $('[name="notice_text"]').val(data.notice_text);
                $('[name="is_default"]').prop('checked', data.is_default == 1);

                if (data.logo_path) {
                    $('#logo_preview').html('<img src="' + data.logo_path + '" height="60">');

                    $('#wechat_qr_preview').html('<img src="' + data.wechat_qr_path + '" height="100">');

### 3.2 账单生成交互

```javascript
// 批量生成账单
$('#btn-batch-create-statement').on('click', function() {
    var checkedIds = [];
    $('input[name="package_ids[]"]:checked').each(function() {
        checkedIds.push($(this).val());
    });
    
    if (checkedIds.length == 0) {
        layer.msg('请选择订单');
        return;
    }
    
    // 验证是否同一客户
    var memberIds = [];
    $('input[name="package_ids[]"]:checked').each(function() {
        var memberId = $(this).data('member-id');
        if (memberIds.indexOf(memberId) == -1) {
            memberIds.push(memberId);
        }
    });
    
    if (memberIds.length > 1) {
        layer.msg('只能选择同一客户的订单');
        return;
    }
    
    // 确认生成
    layer.confirm('确定为选中的 ' + checkedIds.length + ' 个订单生成账单吗？', function(index) {
        $.ajax({
            url: '/store/package/createStatement',
            type: 'POST',
            data: {
                package_ids: checkedIds,
                member_id: memberIds[0],
                start_date: $('#start_date').val(),
                end_date: $('#end_date').val()
            },
            success: function(res) {
                if (res.code == 1) {
                    layer.msg('账单生成成功');
                    layer.close(index);
                    location.reload();
                } else {
                    layer.msg(res.msg);
                }
            }
        });
    });
});
```

function apiRequest(url, data, method) {
    method = method || 'GET';

        data: data,
        dataType: 'json',

        complete: function() {

    });
