## 一、功能概述

账单生成是核心功能，包括：
1. 待出账订单查询
2. 账单记录创建
3. 订单绑定
4. 单价计算（支持5种计价方式）
5. 账单汇总
6. Excel生成

## 二、后端开发

### 2.1 模型层（Statement.php）

**文件路径**：`source/application/common/model/Statement.php`

```php
<?php
namespace app\common\model;

use think\Model;

class Statement extends Model
{
    protected $name = 'statement';
    protected $pk = 'id';
    
    // 账单状态常量
    const STATUS_DRAFT = 1;  // 草稿
    const STATUS_CONFIRMED = 2;  // 已确认
    const STATUS_SENT = 3;  // 已发送
    
    // 支付状态常量
    const PAY_STATUS_UNPAID = 1;  // 未支付
    const PAY_STATUS_PAID = 2;  // 已支付
    
    /**
     * 关联集运订单
     */
    public function packages()
    {
        return $this->hasMany('Package', 'statement_id', 'id');
    }
    
    /**
     * 生成账单编号
     */
    public static function generateStatementNo()
    {
        $date = date('Ymd');
        $prefix = 'ST' . $date;
        
        // 查询当天最大流水号
        $lastNo = self::where('statement_no', 'like', $prefix . '%')
            ->order('id', 'desc')
            ->value('statement_no');
        
        if ($lastNo) {
            $sequence = intval(substr($lastNo, -4)) + 1;
        } else {
            $sequence = 1;
        }
        
        return $prefix . str_pad($sequence, 4, '0', STR_PAD_LEFT);
    }
}
```

### 2.2 服务层（StatementService.php）

**文件路径**：`source/application/common/service/StatementService.php`

```php
<?php
namespace app\common\service;

use app\common\model\Statement;
use app\common\model\Package;
use app\common\model\FinanceConfig;
use think\Db;

class StatementService
{
    private $wxapp_id;
    
    public function __construct($wxapp_id = null)
    {
        $this->wxapp_id = $wxapp_id ?: helper::getWxappId();
    }
    
    /**
     * 查询待出账订单
     */
    public function getUnbilledPackages($params)
    {
        $model = Package::where('is_delete', 0)
            ->where('statement_id', null)
            ->where('wxapp_id', $this->wxapp_id);
        
        // 客户筛选
        if (!empty($params['member_id'])) {
            $model->where('member_id', $params['member_id']);
        }
        
        // 日期范围
        if (!empty($params['start_date'])) {
            $model->where('entering_warehouse_time', '>=', $params['start_date']);
        }
        if (!empty($params['end_date'])) {
            $model->where('entering_warehouse_time', '<=', $params['end_date']);
        }
        
        // 可选：物流状态过滤
        if (!empty($params['status'])) {
            $model->where('status', '>=', $params['status']);
        }
        
        return $model->order('entering_warehouse_time', 'asc')->select();
    }
    
    /**
     * 生成账单
     */
    public function createStatement($data)
    {
        $packageIds = $data['package_ids'];
        $memberId = $data['member_id'];
        
        // 验证订单
        $packages = Package::where('id', 'in', $packageIds)
            ->where('member_id', $memberId)
            ->where('statement_id', null)
            ->where('is_delete', 0)
            ->select();
        
        if (count($packages) != count($packageIds)) {
            throw new \Exception('部分订单不可用或已归档');
        }
        
        // 获取配置
        $priceConfig = FinanceConfig::getEffectivePrice($memberId);
        $template = FinanceConfig::getDefaultTemplate();
        
        // 计算金额
        $result = $this->calculateAmount($packages, $priceConfig);
        $packages = $result['packages'];
        $totalAmount = $result['total_amount'];
        $totalWeight = $result['total_weight'];
        
        // 开启事务
        Db::startTrans();
        try {
            // 1. 创建账单记录
            $statementNo = Statement::generateStatementNo();
            $statement = Statement::create([
                'statement_no' => $statementNo,
                'member_id' => $memberId,
                'member_name' => $packages[0]['member_name'],
                'start_date' => $data['start_date'],
                'end_date' => $data['end_date'],
                'total_packages' => count($packages),
                'total_weight' => $totalWeight,
                'total_amount' => $totalAmount,
                'status' => Statement::STATUS_DRAFT,
                'pay_status' => Statement::PAY_STATUS_UNPAID,
                'created_by' => $data['created_by'] ?? 0,
                'wxapp_id' => $this->wxapp_id
            ]);
            
            // 2. 绑定订单
            Package::where('id', 'in', $packageIds)
                ->update(['statement_id' => $statement['id']]);
            
            // 3. 生成Excel
            $excelService = new ExcelService();
            $excelPath = $excelService->generateStatementExcel($statement['id'], $packages, $template);
            
            // 4. 更新Excel路径
            $statement->save(['excel_path' => $excelPath]);
            
            Db::commit();
            
            return $statement;
            
        } catch (\Exception $e) {
            Db::rollback();
            throw $e;
        }
    }
    
    /**
     * 计算账单金额（支持多种计价方式）
     */
    private function calculateAmount($packages, $priceConfig)
    {
        $totalAmount = 0;
        $totalWeight = 0;
        $priceType = $priceConfig['price_type'];
        
        // 如果是阶梯价格（按总重量），需要先计算总重量
        if ($priceType == FinanceConfig::PRICE_TYPE_TIER) {
            $totalWeight = array_sum(array_column($packages, 'weight'));
            $unitPrice = $this->calculateTierPrice($totalWeight, $priceConfig);
            
            // 所有包裹使用同一单价
            foreach ($packages as &$package) {
                $package['unit_price'] = $unitPrice;
                $package['amount'] = $package['weight'] * $unitPrice;
                $totalAmount += $package['amount'];
            }
        } else {
            // 其他方式按单个包裹计算
            foreach ($packages as &$package) {
                $unitPrice = $this->calculateUnitPrice($package, $priceConfig);
                $package['unit_price'] = $unitPrice;
                $package['amount'] = $package['weight'] * $unitPrice;
                $totalAmount += $package['amount'];
                $totalWeight += $package['weight'];
            }
        }
        
        return [
            'packages' => $packages,
            'total_amount' => round($totalAmount, 2),
            'total_weight' => round($totalWeight, 3)
        ];
    }
    
    /**
     * 计算单价
     */
    private function calculateUnitPrice($package, $priceConfig)
    {
        $priceType = $priceConfig['price_type'];
        
        switch ($priceType) {
            case FinanceConfig::PRICE_TYPE_FIXED: // 固定单价
                return $priceConfig['unit_price'];
                
            case FinanceConfig::PRICE_TYPE_LINE: // 按线路
                $lines = json_decode($priceConfig['price_line_json'], true);
                $lineId = $package['line_id'];
                
                foreach ($lines['lines'] as $line) {
                    if ($line['line_id'] == $lineId) {
                        return $line['price'];
                    }
                }
                return $lines['default_price'] ?? 46.00;
                
            case FinanceConfig::PRICE_TYPE_RANGE: // 按重量区间
                $ranges = json_decode($priceConfig['price_range_json'], true);
                $weight = $package['weight'];
                
                foreach ($ranges['ranges'] as $range) {
                    if ($weight >= $range['min'] && 
                        ($range['max'] === null || $weight < $range['max'])) {
                        return $range['price'];
                    }
                }
                break;
                
            case FinanceConfig::PRICE_TYPE_FORMULA: // 自定义公式
                $formula = json_decode($priceConfig['price_formula'], true);
                return $this->evaluateFormula($formula['formula'], $package);
        }
        
        return $priceConfig['unit_price'] ?? 46.00;
    }
    
    /**
     * 计算阶梯价格
     */
    private function calculateTierPrice($totalWeight, $priceConfig)
    {
        $tiers = json_decode($priceConfig['price_tier_json'], true);
        
        foreach ($tiers['tiers'] as $tier) {
            if ($totalWeight >= $tier['min'] && 
                ($tier['max'] === null || $totalWeight < $tier['max'])) {
                return $tier['price'];
            }
        }
        
        return $priceConfig['unit_price'] ?? 46.00;
    }
    
    /**
     * 评估自定义公式
     */
    private function evaluateFormula($formula, $package)
    {
        // 替换变量
        $variables = [
            '{weight}' => $package['weight'],
            '{volume}' => $package['volume'] ?? 0,
            '{quantity}' => $package['num'] ?? 1,
            '{line_id}' => $package['line_id'] ?? 0,
            '{base_price}' => $package['free'] ?? 0,
        ];
        
        $expression = str_replace(array_keys($variables), array_values($variables), $formula);
        
        // 安全评估（使用表达式解析库）
        try {
            // 这里需要引入 symfony/expression-language 或类似库
            // composer require symfony/expression-language
            $expressionLanguage = new \Symfony\Component\ExpressionLanguage\ExpressionLanguage();
            $result = $expressionLanguage->evaluate($expression);
            return round($result, 2);
        } catch (\Exception $e) {
            // 公式错误，返回默认值
            return 46.00;
        }
    }
}
```
